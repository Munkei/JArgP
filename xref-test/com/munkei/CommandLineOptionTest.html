32  public class CommandLineOptionTest {
33  
34    @Test
35    public void test() {
36      for (Field field : TestSubject.class.getFields()) {
37        if (field.isAnnotationPresent(CommandLineOption.class)) {
38          CommandLineOption option = field.getAnnotation(CommandLineOption.class);
39          assertEquals(2, option.names().length);
40          assertArrayEquals(new String[]{ "name1",
41                                          "name2" },
42                            option.names());
43          assertEquals("default", option.defaultValue());
44          assertEquals("description", option.description());
45          assertEquals(true, option.opposite());
46          assertEquals("pattern", option.pattern());
47          assertEquals("setField", option.setter());
48        }
49      }
50    }
51  
52    public class TestSubject {
53  
54      @SuppressWarnings("PublicField")
55      @CommandLineOption(
56        names = { "name1",
57                  "name2" },
58        defaultValue = "default",
59        description = "description",
60        opposite = true,
61        pattern = "pattern",
62        setter = "setField")
63      public String field;
64  
65    }
